  let selected = null;

  // populate grid
  amounts.forEach(([amt, bonus]) => {
    const btn = document.createElement('button');
    btn.type = 'button';
    btn.className = 'amount-box';
    btn.innerHTML = `<div class="amt">KSH ${amt.toLocaleString()}</div>` +
                    (bonus ? `<div class="bonus">+KSH ${bonus.toLocaleString()}</div>` : `<div style="height:18px"></div>`);
    btn.addEventListener('click', () => selectAmount(amt, bonus, btn));
    amountGrid.appendChild(btn);
  });

  function clearSelection(){
    document.querySelectorAll('.amount-box').forEach(b => b.classList.remove('selected'));
    selected = null;
    selectedAmountEl.textContent = '—';
  }

  function selectAmount(amt, bonus, btn){
    if (customInput) customInput.value = '';
    document.querySelectorAll('.amount-box').forEach(b => b.classList.remove('selected'));
    if (btn) btn.classList.add('selected');
    selected = { amt: Number(amt), bonus: Number(bonus) };
    selectedAmountEl.textContent = `KSH ${selected.amt.toLocaleString()} ${selected.bonus ? '(+'+selected.bonus.toLocaleString()+' bonus)' : ''}`;
  }

  if (customInput) {
    customInput.addEventListener('input', () => {
      const v = Number(customInput.value);
      document.querySelectorAll('.amount-box').forEach(b => b.classList.remove('selected'));
      if (!v) { selected = null; selectedAmountEl.textContent = '—'; return; }
      selected = { amt: v, bonus: 0 };
      selectedAmountEl.textContent = `KSH ${v.toLocaleString()}`;
    });
  }

  if (goToPay) {
    goToPay.addEventListener('click', () => {
      const provider = providerSelect.value;
      const amount = selected ? selected.amt : Number(customInput.value);
      if (!amount || amount < 200) { alert('Enter or select an amount of at least KSH 200.'); return; }
      payAmount.textContent = `KSH ${amount.toLocaleString()}`;
      payProvider.textContent = provider.toUpperCase();
      const total = amount + (selected && selected.bonus ? selected.bonus : 0);
      payTotal.textContent = `KSH ${total.toLocaleString()}`;
      // update till based on provider
      if (provider === 'mpesa') {
        tillNumber.textContent = '6984245';
        tillLabel.textContent = 'Buy goods';
      } else {
        tillNumber.textContent = '0792572591';
        tillLabel.textContent = 'Send money';
      }
      selectPage.classList.remove('active');
      payPage.classList.add('active');
      window.scrollTo(0,0);
    });
  }

  if (confirmBtn) {
    confirmBtn.addEventListener('click', () => {
      const provider = providerSelect.value;
      const amount = selected ? selected.amt : Number(customInput.value);
      const id = txId.value.trim();
      if (!amount || amount < 200) { alert('Invalid amount.'); return; }
      if (!id) { alert('Please enter transaction ID.'); return; }
      showSuccess(`KSH ${amount.toLocaleString()} received via ${provider.toUpperCase()}. Transaction ID ${id}.`);
    });
  }

  function showSuccess(message){
    document.getElementById('successMsg').textContent = message;
    overlay.classList.add('show');
  }

    doneBtn.addEventListener('click', () => {
      overlay.classList.remove('show');
